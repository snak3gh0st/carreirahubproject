Conventions to follow (per D-01, D-03, and existing schema patterns):
- IDs use `@id @default(cuid())`
- All FK columns have a matching `@@index([fkColumn])`
- Table names use `@@map("snake_case")`

- Program type and session type are plain `String` fields (NOT Prisma enums) per D-03 / D-12

```prisma
model MentorshipPhase {
  id        String   @id @default(cuid())
  key       String   @unique
  label     String
  sortOrder Int      @unique
  slaDays   Int
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  enrollments        MentorshipEnrollment[] @relation("CurrentPhaseEnrollments")
  transitionsFrom    PhaseTransition[]      @relation("TransitionFrom")
  transitionsTo      PhaseTransition[]      @relation("TransitionTo")

  @@index([sortOrder])
  @@map("mentorship_phases")
}

model MentorshipEnrollment {
  id           String    @id @default(cuid())
  programType  String    // "PASS" | "ADVANCED" (per D-03)
  status       String    @default("ACTIVE") // "ACTIVE" | "COMPLETED" | "PAUSED"
  startDate    DateTime
  endDate      DateTime?
  createdAt    DateTime  @default(now())
  updatedAt    DateTime  @updatedAt

  customerId      String
  customer        Customer         @relation(fields: [customerId], references: [id])
  assignedToId    String
  assignedTo      User             @relation("AssignedEnrollments", fields: [assignedToId], references: [id])
  currentPhaseId  String?
  currentPhase    MentorshipPhase? @relation("CurrentPhaseEnrollments", fields: [currentPhaseId], references: [id])

  sessions    MentorshipSession[]
  transitions PhaseTransition[]

  @@index([customerId])
  @@index([assignedToId])
  @@index([currentPhaseId])
  @@index([status])
  @@map("mentorship_enrollments")
}

model MentorshipSession {
  id           String   @id @default(cuid())
  sessionType  String   // controlled values per D-12
  sessionDate  DateTime
  notes        String?
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  enrollmentId String
  enrollment   MentorshipEnrollment @relation(fields: [enrollmentId], references: [id])
  conductorId  String
  conductor    User   @relation("ConductedSessions", fields: [conductorId], references: [id])

  @@index([enrollmentId])
  @@index([conductorId])
  @@index([sessionDate])
  @@map("mentorship_sessions")
}

model PhaseTransition {
  id          String   @id @default(cuid())
  createdAt   DateTime @default(now())

  enrollmentId    String
  enrollment      MentorshipEnrollment @relation(fields: [enrollmentId], references: [id])
  fromPhaseId     String?
  fromPhase       MentorshipPhase?     @relation("TransitionFrom", fields: [fromPhaseId], references: [id])
  toPhaseId       String
  toPhase         MentorshipPhase      @relation("TransitionTo", fields: [toPhaseId], references: [id])
  triggeredById   String
  triggeredBy     User                 @relation("TriggeredTransitions", fields: [triggeredById], references: [id])

  @@index([enrollmentId])
  @@index([fromPhaseId])
  @@index([toPhaseId])
  @@index([triggeredById])
  @@map("phase_transitions")
}
```

**Step A — Create `prisma/seed.ts`** (file does not exist yet):

```typescript
import { PrismaClient } from "@prisma/client";

const prisma = new PrismaClient();

const PHASES = [
  { sortOrder: 1,  key: "bastao",          label: "Passagem de Bastão", slaDays: 3  },
  { sortOrder: 2,  key: "cadastro",         label: "Cadastro",           slaDays: 3  },
  { sortOrder: 3,  key: "teste_de_ingles",  label: "Teste de Inglês",    slaDays: 7  },
  { sortOrder: 4,  key: "onboarding",       label: "Onboarding",         slaDays: 7  },
  { sortOrder: 5,  key: "board",            label: "Board",              slaDays: 7  },
  { sortOrder: 6,  key: "bussola",          label: "Bússola",            slaDays: 14 },
  { sortOrder: 7,  key: "raio_x",           label: "Raio X",             slaDays: 14 },
  { sortOrder: 8,  key: "material",         label: "Material",           slaDays: 21 },
  { sortOrder: 9,  key: "devolutiva",       label: "Devolutiva",         slaDays: 7  },
  { sortOrder: 10, key: "ongoing",          label: "Ongoing",            slaDays: 60 },
  { sortOrder: 11, key: "renovacao",        label: "Renovação",          slaDays: 14 },
];

async function main() {
  console.log("Seeding mentorship phases...");
  for (const phase of PHASES) {
    await prisma.mentorshipPhase.upsert({
      where: { key: phase.key },
      update: { label: phase.label, sortOrder: phase.sortOrder, slaDays: phase.slaDays },
      create: phase,
    });
  }
  console.log(`Seeded ${PHASES.length} mentorship phases.`);
}

main()
  .catch((e) => {
    console.error(e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
```
