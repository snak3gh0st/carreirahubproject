    ```typescript
    export type CEFRLevel = 'A1' | 'A2' | 'B1' | 'B2' | 'C1' | 'C2';

    export interface BankQuestion {
      id: string;           // format: "{level}_{skill}_{number}" e.g. "a1_gram_01"

      options: string[];    // always 4 options
      correctIndex: number; // 0-based

      explanation?: string; // future use
    }

    export interface ClientQuestion {
      id: string;
      section: number;
      question: string;
      options: string[];
      passage?: string;
    }

    export interface TestResult {
      sectionScores: number[];  // [s1, s2, s3, s4, s5]
      sectionMaxes: number[];   // [max1, max2, max3, max4, max5]
      totalScore: number;
      percentage: number;       // 0-100
      cefrLevel: string;
      displayLevel: string;
      displayLevelPt: string;
    }
    ```

    **index.ts** — Implement:
    1. `getAllQuestions()` — Returns all questions from all level files combined. Import from questions-a1.ts through questions-c2.ts (stub imports for b1-c2 that import empty arrays for now, will be filled in Plan 02).
    2. `getQuestionsByIds(ids: string[])` — Returns BankQuestion[] for the given IDs from the full bank. Build a Map for O(1) lookup.
    3. `toClientQuestion(q: BankQuestion): ClientQuestion` — Strip correctIndex and explanation. Only include passage if present.
    4. `selectRandom<T>(pool: T[], count: number): T[]` — Fisher-Yates partial shuffle. Copy array first, swap from index 0 to count-1, return slice(0, count).
    5. `generateTest(seenIds: Set<string>, questionsPerSection?: number): BankQuestion[]` — Default 5 questions per section. For each section 1-5: filter bank by section, exclude seenIds, if remaining < needed then reset pool (use full section pool), call selectRandom. Combine all sections and return.
    6. `scoreAnswers(answers: Record<string, number>, questionIds: string[]): TestResult` — Look up each question by ID in the bank. Group by section. Count correct per section. Apply contiguous pass algorithm with 60% threshold: `Math.ceil(sectionCount * 0.6)`. Map to CEFR level: 0 passed=A1, 1=A2, 2=B1, 3=B2, 4=C1, 5 with perfect section 5=C2 else C1. Use DISPLAY_LEVELS from types.
