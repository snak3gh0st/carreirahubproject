Add contract trigger logic:

```typescript
import { docusignService } from './docusign.service';
import { prisma } from '@/lib/db';
import { integrationLogger } from '@/lib/utils/logger';

interface ContractTriggerOptions {
  invoiceId: string;
  delayMinutes?: number; // Default: 5-10 minutes random
}

/**
 * Trigger contract generation for an invoice
 * 
 * Business Rules:
 * - Only generate contract for FIRST invoice in an invoice series
 * - Invoice series identified by prefix (e.g., "JD-" for João)
 * - Skip if customer already has contract for same series
 * - Delay 5-10 minutes after invoice is sent
 * - Alert commercial team when contract is sent
 * 
 * @param options Invoice ID and optional delay
 */
export async function triggerContractGeneration(options: ContractTriggerOptions): Promise<void> {
  const { invoiceId, delayMinutes } = options;
  
  try {
    // Fetch invoice with related data
    const invoice = await prisma.invoice.findUnique({
      where: { id: invoiceId },
      include: {
        customer: true,
        deal: true,
      },
    });
    
    if (!invoice) {
      console.error(`[CONTRACT] Invoice ${invoiceId} not found`);
      return;
    }
    
    if (!invoice.customer) {
      console.error(`[CONTRACT] Invoice ${invoiceId} has no customer`);
      return;
    }
    
    // Extract invoice series from invoice number (e.g., "JD-2024-01-001" → "JD")
    const invoiceSeries = extractInvoiceSeries(invoice.invoiceNumber);
    
    if (!invoiceSeries) {
      console.log(`[CONTRACT] Skipping contract for invoice ${invoice.invoiceNumber} - no series detected`);
      return;
    }
    
    console.log(`[CONTRACT] Invoice series: ${invoiceSeries}`);
    
    // Check if this is the first invoice for this customer in this series
    const isFirstInvoice = await isFirstInvoiceInSeries(invoice.customerId, invoiceSeries, invoice.id);
    
    if (!isFirstInvoice) {
      console.log(`[CONTRACT] Skipping contract - not first invoice for customer ${invoice.customer.name} in series ${invoiceSeries}`);
      return;
    }
    
    console.log(`[CONTRACT] First invoice detected for series ${invoiceSeries}`);
    
    // Check if customer already has a contract for this series
    const existingContract = await prisma.contract.findFirst({
      where: {
        customerId: invoice.customerId,
        invoiceSeries: invoiceSeries,
        status: { notIn: ['declined', 'voided'] }, // Exclude declined/voided
      },
    });
    
    if (existingContract) {
      console.log(`[CONTRACT] Customer ${invoice.customer.name} already has contract for series ${invoiceSeries}`);
      
      // Alert commercial team about duplicate attempt
      await alertCommercialDuplicateContract(invoice, invoiceSeries, existingContract);
      return;
    }
    
    // Calculate delay (5-10 minutes random, or specified)
    const delay = delayMinutes
      ? delayMinutes * 60 * 1000
      : (5 + Math.random() * 5) * 60 * 1000; // 5-10 min
    
    console.log(`[CONTRACT] Scheduling contract generation in ${Math.round(delay / 60000)} minutes`);
    
    // Schedule contract generation
    setTimeout(async () => {
      await generateContract(invoice, invoiceSeries);
    }, delay);
    
  } catch (error) {
    console.error('[CONTRACT] Failed to trigger contract generation:', error);
    await integrationLogger.logError(
      'docusign',
      'triggerContractGeneration',
      error as Error,
      {
        errorCode: 'CONTRACT_TRIGGER_FAILED',
        category: 'unknown',
      },
      { invoiceId }
    );
  }
}

/**
 * Extract invoice series from invoice number
 * Examples:
 *   "JD-2024-01-001" → "JD"
 *   "MS-2024-12-005" → "MS"
 *   "CUST-2024-01-001" → "CUST"
 */
function extractInvoiceSeries(invoiceNumber: string | null): string | null {
  if (!invoiceNumber) return null;
  
  // Match pattern: <LETTERS>-<YEAR>-<MONTH>-<NUMBER>
  const match = invoiceNumber.match(/^([A-Z]+)-\d{4}-\d{2}-\d+$/);
  return match ? match[1] : null;
}

/**
 * Check if this is the first invoice for customer in this series
 */
async function isFirstInvoiceInSeries(
  customerId: string,
  invoiceSeries: string,
  currentInvoiceId: string
): Promise<boolean> {
  const count = await prisma.invoice.count({
    where: {
      customerId: customerId,
      invoiceNumber: { startsWith: `${invoiceSeries}-` },
      id: { not: currentInvoiceId }, // Exclude current invoice
    },
  });
  
  return count === 0;
}

/**
 * Generate contract for invoice
 */
async function generateContract(invoice: any, invoiceSeries: string): Promise<void> {
  try {
    console.log(`[CONTRACT] Generating contract for invoice ${invoice.invoiceNumber}`);
    
    // Create envelope in DocuSign
    const envelopeId = await docusignService.createEnvelopeFromTemplate(
      {
        id: invoice.id,
        invoiceNumber: invoice.invoiceNumber,
        amount: invoice.amount,
        dueDate: invoice.dueDate,
        deal: invoice.deal,
      },
      {
        id: invoice.customer.id,
        name: invoice.customer.name,
        email: invoice.customer.email,
        phone: invoice.customer.phone,
      }
    );
    
    console.log(`[CONTRACT] Envelope created: ${envelopeId}`);
    
    // Create contract record in database
    const contract = await prisma.contract.create({
      data: {
        customerId: invoice.customerId,
        dealId: invoice.dealId,
        invoiceId: invoice.id,
        envelopeId: envelopeId,
        invoiceSeries: invoiceSeries,
        status: 'sent',
        sentAt: new Date(),
      },
    });
    
    console.log(`[CONTRACT] Contract record created: ${contract.id}`);
    
    // Alert commercial team
    await alertCommercialContractSent(invoice, contract);
    
    // Log success
    await integrationLogger.logSuccess(
      'docusign',
      'contract_generated',
      { invoiceId: invoice.id, envelopeId, contractId: contract.id }
    );
    
  } catch (error) {
    console.error('[CONTRACT] Failed to generate contract:', error);
    
    // Log error but DO NOT rollback invoice
    await integrationLogger.logError(
      'docusign',
      'contract_generation',
      error as Error,
      {
        errorCode: 'CONTRACT_GENERATION_FAILED',
        category: 'transient',
        recovery: 'retry',
      },
      { invoiceId: invoice.id }
    );
    
    // Alert tech team
    await alertTechContractFailed(invoice, error as Error);
  }
}
```

```prisma
model Contract {
```

```typescript
function getContractStatusColor(status: string): string {
  switch (status) {
    case 'signed':
    case 'completed':
      return 'bg-green-100 text-green-800';
    case 'sent':
    case 'delivered':
      return 'bg-blue-100 text-blue-800';
    case 'viewed':
      return 'bg-purple-100 text-purple-800';
    case 'declined':
      return 'bg-red-100 text-red-800';
    case 'voided':
    case 'expired':
      return 'bg-gray-100 text-gray-800';
    default:
      return 'bg-gray-100 text-gray-800';
  }
}

function formatContractStatus(status: string): string {
  return status.charAt(0).toUpperCase() + status.slice(1);
}

async function downloadContract(contractId: string) {
  try {
    const response = await fetch(`/api/contracts/${contractId}/download`);
    const data = await response.json();
    
    if (data.url) {
      window.open(data.url, '_blank');
    } else {
      alert('Failed to generate download link');
    }
  } catch (error) {
    console.error('Download failed:', error);
    alert('Failed to download contract');
  }
}
```
