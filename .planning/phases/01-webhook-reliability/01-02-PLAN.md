**Context from Plan 01-01:**
- WebhookEvent table exists with event_id field (indexed)
- handleWebhookWithRetry middleware persists all incoming webhooks
- All webhooks are logged with service and event_id

**Deduplication strategy:**
- Use event_id from webhook payload as idempotency key
- Check WebhookEvent table for existing event with same (service + event_id)
- If exists and processed: return 200 OK immediately (skip processing)
- If exists and failed: continue with retry logic
- If new: process normally

**Webhook provider event ID fields:**
- Pipedrive: body.meta.v (webhook version-specific ID) or body.current.id
- QuickBooks: body.eventNotifications[].realmId + dataChangeEvent.entities[].id
- Stripe: body.id (event ID)
- DocuSign: body.eventId or envelopeId
- Twilio: body.MessageSid
- RetellAI: body.call_id

**Constraints:**
- Vercel serverless (10s timeout)
- Must not break existing webhook functionality
- Deduplication check must be fast (database query)
</context>

```typescript
export function extractEventId(service: string, payload: any): string | null {
  switch (service) {
    case "pipedrive":
      // V2 webhooks: body.meta.v or body.data.id
      // V1 webhooks: body.current.id
      return payload?.meta?.v || payload?.data?.id || payload?.current?.id || null;

    case "quickbooks":
      // Combine realmId + entity ID for uniqueness
      const realmId = payload?.eventNotifications?.[0]?.realmId;
      const entityId = payload?.eventNotifications?.[0]?.dataChangeEvent?.entities?.[0]?.id;
      return realmId && entityId ? `${realmId}-${entityId}` : null;

    case "stripe":
      // Stripe events have top-level id field
      return payload?.id || null;

    case "docusign":
      // Use eventId or envelopeId as fallback
      return payload?.eventId || payload?.data?.envelopeId || null;

    case "twilio":
      // Use MessageSid for WhatsApp messages
      return payload?.MessageSid || null;

    case "retell":
      // Use call_id from RetellAI
      return payload?.call_id || null;

    default:
      // Fallback: generate hash of payload if no event ID available
      // Use crypto to create deterministic hash
      const hash = createHash("sha256").update(JSON.stringify(payload)).digest("hex");
      return hash;
  }
}
```
