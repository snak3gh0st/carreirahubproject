**Step B — Create `app/ops/pipeline/usePipelineData.ts`**:

This hook owns all data fetching and mutation for the board. It exports:
- `usePipelineData()` — React Query useQuery wrapping GET /api/ops/pipeline
- `useAdvancePhase()` — React Query useMutation wrapping POST /api/ops/enrollments/[id]/advance, with optimistic updates

For the optimistic update in `useAdvancePhase`:
- `onMutate`: capture the current query snapshot via `queryClient.getQueryData(["pipeline"])`, then call `queryClient.setQueryData(["pipeline"], ...)` to move the enrollment from its current phase column to the target phase column in the local cache
- `onError`: restore the captured snapshot with `queryClient.setQueryData(["pipeline"], snapshot)`
- `onSettled`: call `queryClient.invalidateQueries({ queryKey: ["pipeline"] })` to re-sync with server

Type the query data as `PhaseWithEnrollments[]` using the interface defined in the context interfaces block above. Export the interface types from this file so PipelineBoard.tsx and StudentCard.tsx can import them.

import { getServerSession } from "next-auth";
import { authOptions } from "@/lib/auth";
import { redirect } from "next/navigation";
import { Suspense } from "react";
import { PipelineBoard } from "./PipelineBoard";

export const metadata = { title: "Pipeline | Ops Hub" };

export default async function PipelinePage() {
  const session = await getServerSession(authOptions);
  if (!session?.user) redirect("/ops/login");
  const role = (session.user as any).role as string;
  if (role !== "ADMIN" && role !== "OPERATIONAL") redirect("/ops");

  const currentUserId = (session.user as any).id as string;
  const currentUserName = (session.user as any).name as string ?? "";

  return (
    <div className="p-6">
      <div className="mb-6">
        <h1 className="text-2xl font-display font-bold text-brand-verde">Pipeline</h1>
        <p className="text-sm text-gray-500 mt-1">Jornada dos alunos por fase</p>
      </div>
      <Suspense fallback={<PipelineSkeleton />}>
        <PipelineBoard currentUserId={currentUserId} currentUserName={currentUserName} />
      </Suspense>
    </div>
  );
}

function PipelineSkeleton() {
  return (
    <div className="flex gap-4 overflow-x-auto pb-4">
      {Array.from({ length: 11 }).map((_, i) => (
        <div key={i} className="flex-shrink-0 w-52 h-96 bg-gray-100 rounded-xl animate-pulse" />
      ))}
    </div>
  );
}

const lastTransition = enrollment.transitions[0];
