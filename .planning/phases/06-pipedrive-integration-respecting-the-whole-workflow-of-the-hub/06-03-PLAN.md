```typescript
async createDeal(data: {
  title: string;
  person_id: number;
  value: number;
  currency: string;
}): Promise<{ id: number; title: string; value: number }> {
  return this.circuitBreaker.execute(async () => {
    const response = await fetch(
      `${this.baseURL}/deals`,
      {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${this.apiToken}`,
        },
        body: JSON.stringify({
          title: data.title,
          person_id: data.person_id,
          value: data.value,
          currency: data.currency
        }),
      }
    );

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`Pipedrive API error: ${response.status} - ${error}`);
    }

    const result = await response.json();
    
    if (!result.success) {
      throw new Error(`Pipedrive createDeal failed: ${result.error}`);
    }

    return result.data;
  });
}
```

```typescript
  value?: number;
  currency?: string;
  status?: string;
}): Promise<{ id: number; value: number }> {
  return this.circuitBreaker.execute(async () => {
    const response = await fetch(
      `${this.baseURL}/deals/${dealId}`,
      {
        method: "PUT",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${this.apiToken}`,
        },

      }
    );

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`Pipedrive API error: ${response.status} - ${error}`);
    }

    const result = await response.json();
    
    if (!result.success) {
      throw new Error(`Pipedrive updateDeal failed: ${result.error}`);
    }

    return result.data;
  });
}
```
