```typescript
async markDealAsWon(dealId: number): Promise<void> {
  return this.circuitBreaker.execute(async () => {
    const response = await fetch(
      `${this.baseURL}/deals/${dealId}`,
      {
        method: "PUT",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${this.apiToken}`,
        },
        body: JSON.stringify({
          status: "won"
        }),
      }
    );

    if (!response.ok) {
      const error = await response.text();
      throw new Error(`Pipedrive API error: ${response.status} - ${error}`);
    }

    const result = await response.json();
    
    if (!result.success) {
      throw new Error(`Pipedrive markDealAsWon failed: ${result.error}`);
    }
  });
}
```

Method must:
- Accept dealId (number - Pipedrive's deal ID, not Hub's UUID)
- Update deal status to "won" via Pipedrive API
- Use circuit breaker for resilience
