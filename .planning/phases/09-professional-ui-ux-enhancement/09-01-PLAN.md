**Goal:** Load Inter (primary font) and JetBrains Mono (monospace) from Google Fonts CDN with all required weights.

**Add font import at the TOP of app/globals.css (before @tailwind directives):**

```css
/* Google Fonts - Inter (Primary) and JetBrains Mono (Monospace) */
@import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700&family=JetBrains+Mono:wght@400&display=swap');
```

**Why at the top:** Font imports must come before @tailwind directives to ensure fonts are available when Tailwind processes the CSS.

**Why these weights:**
- Inter 400 (regular) - body text
- Inter 500 (medium) - labels, H4-H6
- Inter 600 (semibold) - headings H1-H3, financial numbers
- Inter 700 (bold) - emphasis
- JetBrains Mono 400 - IDs, codes, technical data

**Why display=swap:** Prevents flash of invisible text (FOIT) by showing fallback font while custom fonts load, improving perceived performance.
  </action>

/**
 * Get color value by path (e.g., 'primary.600', 'success.100')
 */
export function getColor(path: string): string {
  const [category, shade] = path.split('.');
  if (!category || !shade) {
    throw new Error(`Invalid color path: ${path}. Use format 'category.shade' (e.g., 'primary.600')`);
  }
  
  const colorCategory = colors[category as keyof typeof colors];
  if (!colorCategory) {
    throw new Error(`Unknown color category: ${category}`);
  }
  
  const colorValue = colorCategory[shade as keyof typeof colorCategory];
  if (!colorValue) {
    throw new Error(`Unknown shade ${shade} in category ${category}`);
  }
  
  return colorValue;
}

/**
 * Get all shades of a color category as array (useful for charts)
 */
export function getColorScale(category: keyof typeof colors): string[] {
  const colorCategory = colors[category];
  return Object.values(colorCategory);
}
